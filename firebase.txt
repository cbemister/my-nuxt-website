import { db } from '~/plugins/firebase.js'

data() {
  return {
    users: []
  }
},
mounted() {
  db.collection("users").get().then((querySnapshot) => {
    this.users = querySnapshot.docs.map(doc =>
      Object.assign({ id: doc.id }, doc.data())
    )
  })
}
